/* * A simple queue data type. Queues are entirely mutable. The * add and remove methods are simple conveniences, but clients * are free to mess with the queue in any way they like. */ class Queue { constructor() { this.data = [] } add(x) { this.data.push(x) } remove() { return this.data.shift() } } QUnit.test("Trivial queue test", assert => { const q1 = new Queue() const q2 = new Queue() q1.add(0) q1.add(1) q1.add(2) q2.add(3) assert.equal(q1.remove(), 0) assert.equal(q2.remove(), 3) assert.equal(q1.remove(), 1) })
